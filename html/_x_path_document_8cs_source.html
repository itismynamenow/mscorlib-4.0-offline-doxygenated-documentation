144  internal void LoadFromReader(XmlReader reader, XmlSpace space)
145  {
146  if (reader == null)
147  {
148  throw new ArgumentNullException("reader");
149  }
150  IXmlLineInfo xmlLineInfo = reader as IXmlLineInfo;
151  if (xmlLineInfo == null || !xmlLineInfo.HasLineInfo())

153  xmlLineInfo = null;
154  }
155  hasLineInfo = (xmlLineInfo != null);
156  nameTable = reader.NameTable;
157  XPathDocumentBuilder xPathDocumentBuilder = new XPathDocumentBuilder(this, xmlLineInfo, reader.BaseURI, LoadFlags.None);
158  try
159  {
160  bool flag = reader.ReadState == ReadState.Initial;
161  int depth = reader.Depth;
162  string text = nameTable.Get("http://www.w3.org/2000/xmlns/");
163  if (!flag || reader.Read())
164  {
165  while (flag || reader.Depth >= depth)
166  {
167  switch (reader.NodeType)
168  {
169  case XmlNodeType.Element:
170  {
171  bool isEmptyElement = reader.IsEmptyElement;
172  xPathDocumentBuilder.WriteStartElement(reader.Prefix, reader.LocalName, reader.NamespaceURI, reader.BaseURI);
173  while (reader.MoveToNextAttribute())
174  {
175  string namespaceURI = reader.NamespaceURI;
176  if ((object)namespaceURI == text)
177  {
178  if (reader.Prefix.Length == 0)
179  {
180  xPathDocumentBuilder.WriteNamespaceDeclaration(string.Empty, reader.Value);
181  }
182  else
183  {
184  xPathDocumentBuilder.WriteNamespaceDeclaration(reader.LocalName, reader.Value);
185  }
186  }
187  else
188  {
189  xPathDocumentBuilder.WriteStartAttribute(reader.Prefix, reader.LocalName, namespaceURI);
190  xPathDocumentBuilder.WriteString(reader.Value, TextBlockType.Text);
191  xPathDocumentBuilder.WriteEndAttribute();
192  }
193  }
194  if (isEmptyElement)
195  {
196  xPathDocumentBuilder.WriteEndElement(allowShortcutTag: true);
197  }
198  break;
199  }
200  case XmlNodeType.EndElement:
201  xPathDocumentBuilder.WriteEndElement(allowShortcutTag: false);
202  break;
203  case XmlNodeType.Text:
204  case XmlNodeType.CDATA:
205  xPathDocumentBuilder.WriteString(reader.Value, TextBlockType.Text);
206  break;
207  case XmlNodeType.SignificantWhitespace:
208  if (reader.XmlSpace == XmlSpace.Preserve)
209  {
210  xPathDocumentBuilder.WriteString(reader.Value, TextBlockType.SignificantWhitespace);
211  break;
212  }
213  goto case XmlNodeType.Whitespace;
214  case XmlNodeType.Whitespace:
215  if (space == XmlSpace.Preserve && (!flag || reader.Depth != 0))

218  }

222  break;
