85  private void ExportType(TypeMapping mapping)
86  {
87  if (mapping.IsReference || base.ExportedMappings[mapping] != null)
88  {
89  return;
90  }
91  CodeTypeDeclaration codeTypeDeclaration = null;
92  base.ExportedMappings.Add(mapping, mapping);
93  if (mapping is EnumMapping)
94  {
95  codeTypeDeclaration = ExportEnum((EnumMapping)mapping, typeof(SoapEnumAttribute));
96  }
97  else if (mapping is StructMapping)
98  {
99  codeTypeDeclaration = ExportStruct((StructMapping)mapping);
100  }
101  else if (mapping is ArrayMapping)
102  {
103  EnsureTypesExported(((ArrayMapping)mapping).Elements, null);
104  }
105  if (codeTypeDeclaration != null)
106  {
107  codeTypeDeclaration.CustomAttributes.Add(base.GeneratedCodeAttribute);
108  codeTypeDeclaration.CustomAttributes.Add(new CodeAttributeDeclaration(typeof(SerializableAttribute).FullName));
109  if (!codeTypeDeclaration.IsEnum)
110  {
111  codeTypeDeclaration.CustomAttributes.Add(new CodeAttributeDeclaration(typeof(DebuggerStepThroughAttribute).FullName));
112  codeTypeDeclaration.CustomAttributes.Add(new CodeAttributeDeclaration(typeof(DesignerCategoryAttribute).FullName, new CodeAttributeArgument(new CodePrimitiveExpression("code"))));
113  }
114  AddTypeMetadata(codeTypeDeclaration.CustomAttributes, typeof(SoapTypeAttribute), mapping.TypeDesc.Name, Accessor.UnescapeName(mapping.TypeName), mapping.Namespace, mapping.IncludeInSchema);
115  base.ExportedClasses.Add(mapping, codeTypeDeclaration);
116  }

119  private CodeTypeDeclaration ExportStruct(StructMapping mapping)
120  {
121  if (mapping.TypeDesc.IsRoot)
122  {
123  ExportRoot(mapping, typeof(SoapIncludeAttribute));
124  return null;
125  }
126  if (!mapping.IncludeInSchema)
127  {
128  return null;
129  }
130  string name = mapping.TypeDesc.Name;
131  string text = (mapping.TypeDesc.BaseTypeDesc == null) ? string.Empty : mapping.TypeDesc.BaseTypeDesc.Name;
132  CodeTypeDeclaration codeTypeDeclaration = new CodeTypeDeclaration(name);
133  codeTypeDeclaration.IsPartial = base.CodeProvider.Supports(GeneratorSupport.PartialTypes);
134  codeTypeDeclaration.Comments.Add(new CodeCommentStatement(Res.GetString("XmlRemarks"), docComment: true));
135  base.CodeNamespace.Types.Add(codeTypeDeclaration);
136  if (text != null && text.Length > 0)
137  {
138  codeTypeDeclaration.BaseTypes.Add(text);
139  }
140  else
141  {
142  AddPropertyChangedNotifier(codeTypeDeclaration);
143  }
144  codeTypeDeclaration.TypeAttributes |= TypeAttributes.Public;
145  if (mapping.TypeDesc.IsAbstract)
146  {
147  codeTypeDeclaration.TypeAttributes |= TypeAttributes.Abstract;
148  }
149  CodeExporter.AddIncludeMetadata(codeTypeDeclaration.CustomAttributes, mapping, typeof(SoapIncludeAttribute));
150  if (base.GenerateProperties)
151  {
152  for (int i = 0; i < mapping.Members.Length; i++)
153  {
154  ExportProperty(codeTypeDeclaration, mapping.Members[i], mapping.Scope);
155  }
156  }
157  else
158  {
159  for (int j = 0; j < mapping.Members.Length; j++)
160  {
161  ExportMember(codeTypeDeclaration, mapping.Members[j]);
162  }
163  }
164  for (int k = 0; k < mapping.Members.Length; k++)
165  {
166  EnsureTypesExported(mapping.Members[k].Elements, null);
167  }
168  if (mapping.BaseMapping != null)
169  {
170  ExportType(mapping.BaseMapping);
171  }
172  ExportDerivedStructs(mapping);
173  CodeGenerator.ValidateIdentifiers(codeTypeDeclaration);
174  return codeTypeDeclaration;
175  }
