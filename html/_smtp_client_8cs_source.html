636  }

673  mailWriter.Close();

676  {
677  throw exception;
678  }
679  }

681  {
682  if (Logging.On)

684  Logging.Exception(Logging.Web, this, "Send", ex);
685  }

687  {
688  throw;
689  }
690  Abort();
691  if (timedOut)
692  {
693  throw new SmtpException(SR.GetString("net_timeout"));
694  }
695  if (ex is SecurityException || ex is AuthenticationException || ex is SmtpException)
696  {
697  throw;
698  }
699  throw new SmtpException(SR.GetString("SmtpSendMailFailure"), ex);
700  }

708  }
