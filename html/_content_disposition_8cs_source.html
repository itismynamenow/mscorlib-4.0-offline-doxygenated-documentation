335  if (string.IsNullOrEmpty(dispositionType))

338  }
339  if (parameters == null)

343  else
344  {

358  if (disposition[offset++] != '=')

361  }
362  if (!MailBnfHelper.SkipCFWS(disposition, ref offset))
363  {
364  throw new FormatException(SR.GetString("ContentDispositionInvalid"));
365  }

367  if (string.IsNullOrEmpty(text) || string.IsNullOrEmpty(value))
