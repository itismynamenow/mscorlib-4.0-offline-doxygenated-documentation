371  private void ExportTypeMembers(XmlSchemaComplexType type, MemberMapping[] members, string ns)
372  {
373  XmlSchemaGroupBase xmlSchemaGroupBase = new XmlSchemaSequence();
374  foreach (MemberMapping memberMapping in members)
375  {
376  if (memberMapping.Elements.Length != 0)
377  {
378  bool valueTypeOptional = memberMapping.CheckSpecified != 0 || memberMapping.CheckShouldPersist || !memberMapping.TypeDesc.IsValueType;
379  ExportElementAccessors(xmlSchemaGroupBase, memberMapping.Elements, repeats: false, valueTypeOptional, ns);
380  }
381  }
382  if (xmlSchemaGroupBase.Items.Count <= 0)
383  {
384  return;
385  }
386  if (type.ContentModel != null)

390  ((XmlSchemaComplexContentExtension)type.ContentModel.Content).Particle = xmlSchemaGroupBase;
391  return;
392  }
393  if (!(type.ContentModel.Content is XmlSchemaComplexContentRestriction))
394  {
395  throw new InvalidOperationException(Res.GetString("XmlInvalidContent", type.ContentModel.Content.GetType().Name));
396  }
397  ((XmlSchemaComplexContentRestriction)type.ContentModel.Content).Particle = xmlSchemaGroupBase;
