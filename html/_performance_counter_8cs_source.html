542  private void InitializeImpl()
543  {
544  bool lockTaken = false;
545  RuntimeHelpers.PrepareConstrainedRegions();
546  try
547  {
548  Monitor.Enter(InstanceLockObject, ref lockTaken);
549  if (!initialized)
550  {
551  string text = categoryName;
552  string text2 = machineName;
553  if (text == string.Empty)
554  {
555  throw new InvalidOperationException(SR.GetString("CategoryNameMissing"));
556  }
557  if (counterName == string.Empty)
558  {
559  throw new InvalidOperationException(SR.GetString("CounterNameMissing"));
560  }
561  if (ReadOnly)
562  {
563  PerformanceCounterPermission performanceCounterPermission = new PerformanceCounterPermission(PerformanceCounterPermissionAccess.Browse, text2, text);
564  performanceCounterPermission.Demand();
565  if (!PerformanceCounterLib.CounterExists(text2, text, counterName))
566  {
567  throw new InvalidOperationException(SR.GetString("CounterExists", text, counterName));
568  }
569  switch (PerformanceCounterLib.GetCategoryType(text2, text))
570  {
571  case PerformanceCounterCategoryType.MultiInstance:
572  if (string.IsNullOrEmpty(instanceName))
573  {
574  throw new InvalidOperationException(SR.GetString("MultiInstanceOnly", text));
575  }
576  break;
577  case PerformanceCounterCategoryType.SingleInstance:
578  if (!string.IsNullOrEmpty(instanceName))
579  {
580  throw new InvalidOperationException(SR.GetString("SingleInstanceOnly", text));
581  }
582  break;
583  }
584  if (instanceLifetime != 0)
585  {
586  throw new InvalidOperationException(SR.GetString("InstanceLifetimeProcessonReadOnly"));
587  }
588  initialized = true;
589  }
590  else
591  {
592  PerformanceCounterPermission performanceCounterPermission2 = new PerformanceCounterPermission(PerformanceCounterPermissionAccess.Write, text2, text);
593  performanceCounterPermission2.Demand();
594  if (text2 != "." && string.Compare(text2, PerformanceCounterLib.ComputerName, StringComparison.OrdinalIgnoreCase) != 0)
595  {
596  throw new InvalidOperationException(SR.GetString("RemoteWriting"));
597  }
598  SharedUtils.CheckNtEnvironment();
599  if (!PerformanceCounterLib.IsCustomCategory(text2, text))
600  {
601  throw new InvalidOperationException(SR.GetString("NotCustomCounter"));
602  }
603  switch (PerformanceCounterLib.GetCategoryType(text2, text))
604  {
605  case PerformanceCounterCategoryType.MultiInstance:
606  if (string.IsNullOrEmpty(instanceName))
607  {
608  throw new InvalidOperationException(SR.GetString("MultiInstanceOnly", text));
609  }
610  break;
611  case PerformanceCounterCategoryType.SingleInstance:
612  if (!string.IsNullOrEmpty(instanceName))
613  {
614  throw new InvalidOperationException(SR.GetString("SingleInstanceOnly", text));
615  }
616  break;
617  }
618  if (string.IsNullOrEmpty(instanceName) && InstanceLifetime == PerformanceCounterInstanceLifetime.Process)
619  {
620  throw new InvalidOperationException(SR.GetString("InstanceLifetimeProcessforSingleInstance"));
621  }
622  sharedCounter = new SharedPerformanceCounter(text.ToLower(CultureInfo.InvariantCulture), counterName.ToLower(CultureInfo.InvariantCulture), instanceName.ToLower(CultureInfo.InvariantCulture), instanceLifetime);
623  initialized = true;
624  }
625  }
626  }
627  finally
628  {
629  if (lockTaken)
630  {
631  Monitor.Exit(InstanceLockObject);
632  }
633  }
