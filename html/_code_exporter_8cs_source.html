224  {

228  }
229  rootExported = true;

232  {
233  if (!structMapping.ReferencedByElement && structMapping.IncludeInSchema && !structMapping.IsAnonymousType)
234  {
235  CodeAttributeDeclaration codeAttributeDeclaration = new CodeAttributeDeclaration(includeType.FullName);
236  codeAttributeDeclaration.Arguments.Add(new CodeAttributeArgument(new CodeTypeOfExpression(structMapping.TypeDesc.FullName)));
237  includeMetadata.Add(codeAttributeDeclaration);
238  }
239  }
240  Hashtable hashtable = new Hashtable();
241  foreach (TypeMapping typeMapping in Scope.TypeMappings)
242  {
243  if (typeMapping is ArrayMapping)
244  {
245  ArrayMapping arrayMapping = (ArrayMapping)typeMapping;
246  if (ShouldInclude(arrayMapping) && !hashtable.Contains(arrayMapping.TypeDesc.FullName))
247  {
248  CodeAttributeDeclaration codeAttributeDeclaration2 = new CodeAttributeDeclaration(includeType.FullName);
249  codeAttributeDeclaration2.Arguments.Add(new CodeAttributeArgument(new CodeTypeOfExpression(arrayMapping.TypeDesc.FullName)));
250  includeMetadata.Add(codeAttributeDeclaration2);
251  hashtable.Add(arrayMapping.TypeDesc.FullName, string.Empty);
252  EnsureTypesExported(arrayMapping.Elements, arrayMapping.Namespace);
253  }
254  }
