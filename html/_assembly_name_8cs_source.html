517  {

527  return ProcessorArchitecture.IA64;
528  case ImageFileMachine.AMD64:
529  return ProcessorArchitecture.Amd64;

531  if ((pek & PortableExecutableKinds.ILOnly) == PortableExecutableKinds.ILOnly)
