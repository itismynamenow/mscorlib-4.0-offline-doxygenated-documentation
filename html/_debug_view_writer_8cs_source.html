696  {

700  }

705  case ExpressionType.Unbox:
706  case ExpressionType.IsTrue:
707  case ExpressionType.IsFalse:
708  return true;
709  default:
710  {
711  int operatorPrecedence = GetOperatorPrecedence(child);
712  int operatorPrecedence2 = GetOperatorPrecedence(parent);
713  if (operatorPrecedence == operatorPrecedence2)
714  {
715  switch (parent.NodeType)
716  {
717  case ExpressionType.And:
718  case ExpressionType.AndAlso:
719  case ExpressionType.ExclusiveOr:
720  case ExpressionType.Or:

725  case ExpressionType.Multiply:
726  case ExpressionType.MultiplyChecked:
727  return false;
728  case ExpressionType.Divide:
729  case ExpressionType.Modulo:
730  case ExpressionType.Subtract:
731  case ExpressionType.SubtractChecked:
732  {
733  BinaryExpression binaryExpression = parent as BinaryExpression;
734  return child == binaryExpression.Right;
735  }
736  default:
737  return true;
738  }
739  }
740  if (child != null && child.NodeType == ExpressionType.Constant && (parent.NodeType == ExpressionType.Negate || parent.NodeType == ExpressionType.NegateChecked))
741  {
742  return true;
743  }
744  return operatorPrecedence < operatorPrecedence2;
